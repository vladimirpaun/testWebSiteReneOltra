<soapenv:Envelope xmlns:soapenv="http://schemas.xmlsoap.org/soap/envelope/" xmlns:web="http://webservices.webcamp.fr">
  <soapenv:Header/>
  <soapenv:Body>
     <web:cpg_Cancel>
        <RequestCpgCancel_1>
           <idUser>?</idUser>
           <password>?</password>
           <idElement>?</idElement>
           <element>?</element>
           <!--Optional:-->
           <comment>?</comment>
        </RequestCpgCancel_1>
     </web:cpg_Cancel>
  </soapenv:Body>
</soapenv:Envelope>

Mapping

NameTypeMandatoryDefaultRepetitionsComment
idUserstringyes   
passwordstringyes   
idElementstringyes  L'identifiant de l'élément à annuler:
idOrder d'un ORDERS ou d'un ORDERS_U
idPayment : d'un PAYMENT ou d'un PAYMENT_U
elementenumyes  Type de l'élèment à annuler:
order : pour annuler une reservation
payment : pour annuler un paiement
commentstring   Commentaire.
